  <script>
    let dia = 1;

    function avanzar() {
      dia++;

      const texto = document.getElementById("texto");
      const fondo = document.getElementById("fondo");
      const personaje = document.getElementById("personaje");
      const body = document.body;

      if (dia == 2) {
        texto.innerHTML = "Llegas a la escuela. Todo parece normal 😊";
        fondo.src = "imagenes/fondo2.png";
      }

      if (dia == 3) {
        texto.innerHTML = "Sientes que alguien te está mirando...";
      }

      if (dia == 4) {
        body.style.backgroundColor = "#eee";
        texto.innerHTML = "Escuchas un ruido en el pasillo...";
        document.getElementById("sonidoRaro").play();
        fondo.src = "imagenes/fondo3.png";
      }

      if (dia == 5) {
        body.style.backgroundColor = "#ccc";
        document.getElementById("titulo").innerHTML = "¿Estás sola?";
        texto.innerHTML = "...";
      }

      if (dia == 6) {
        body.style.backgroundColor = "#888";
        body.style.color = "#fff";
        texto.innerHTML = "No hay nadie en la escuela...";
      }

      if (dia == 7) {
        body.style.backgroundColor = "black";
        document.getElementById("titulo").innerHTML = "👁️";
        texto.innerHTML = "Te estaba esperando...";
        document.getElementById("sonidoRaro").play();
        personaje.style.transform = "scale(1.2)";
      }

      if (dia > 7) {
        body.innerHTML = "<h1 style='color:red'>NO DEBISTE VENIR HOY</h1>";
      }
    }

    function enviarMensaje() {
      const input = document.getElementById("mensaje");
      const mensaje = input.value.trim();
      if (mensaje) {
        alert("Tu personaje dice: " + mensaje);
        input.value = "";
      }
    }
  </script>
